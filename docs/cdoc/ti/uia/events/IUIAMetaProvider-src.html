    37    package ti.uia.events;
    38    
    39    /*!
    40     *  ======== IUIAMetaProvider ========
    41     *  Interface to identify UIA MetaData providers to the UIAMetaData xml 
    42     *     generation script
    43     *
    44     *  Inheriting from this interface is sufficient to identify to 
    45     *  ti.uia.runtime.UIAMetaData.xs that the module is able to provide metadata
    46     *  for inclusion in the .uia.xml file
    47     */
    48    interface IUIAMetaProvider {
    49    
    50        /*! @_nodoc
    51         * ======== hasMetaData =========
    52         * set to true to indicate that the module has metadata to contribute to the
    53         *    uia.xml file.
    54         */
    55        metaonly config Bool hasMetaData = false;
    56    
    57        /*! @_nodoc
    58         *  ======== writeUIAMetaData ========
    59         *  Writes any UIA metadata required to support the module 
    60         *
    61         *  NOTE: This should only be called by ti.uia.runtime.UIAMetaData.xs in 
    62         *  order to ensure that the UIA xml file has been properly opened and is 
    63         *  ready for writing.     
    64         *  @param(inst) instance of the module to generate data for, null for 
    65         *     module-level info 
    66         *  @param(instNum) the instance number used in .h files
    67         *  @param(indent) the number of spaces to preceed the xml tag with 
    68         */
    69        metaonly function writeUIAMetaData(inst, instNum, indent);
    70    
    71    }
